fastapi==0.115.6

pdfplumber==0.11.5          # digital-PDF text layer + word boxes

# ── OCR engine: DEFAULT (OCR_ENGINE=onnx) ────────────────────────────────────
# RapidOCR runs the PP-OCR models on ONNX Runtime. Bundled models (no download),
# ~50 MB install, and measured on aarch64/CPU: ~8x faster than paddle AND more
# accurate on tilted/blurred photos. Both are open source (Apache-2.0 / MIT).
rapidocr-onnxruntime==1.4.4
onnxruntime==1.27.0

# ── OCR engine: FALLBACK (OCR_ENGINE=classic) ────────────────────────────────
# PaddleOCR. Kept as an escape hatch; not used by default. ~2 GB install.
#
# ⚠️ Engines are MUTUALLY EXCLUSIVE per process — onnxruntime and paddlepaddle
# segfault when loaded together on ARM, so `app/extractor.py` only ever imports
# the configured one. Switch engines via OCR_ENGINE, never at runtime.
#
# NOTE: PaddleOCR 3.x's default model (PP-OCRv6) segfaults on aarch64/ARM, and
# multi-threaded CPU inference does too. `app/config.py` therefore pins
# PP-OCRv5 with cpu_threads=1 — both overridable by env var.
paddlepaddle==3.2.2
paddleocr==3.7.0

# OPTIONAL — PaddleOCR-VL local vision-language OCR engine (OCR_ENGINE=vl).

# ⚠️ Segfaults on aarch64/CPU inference — x86 / GPU hosts only. NOT required for
